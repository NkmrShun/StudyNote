## PHP实现二叉树 ##
### 节点 ###
```
class TreeNode
{
    /**
     * 节点数据
     */
	public $data;
	
    /**
     * 左节点
     */
	public $left;
	
    /**
     * 右节点
     */
	public $right;
	
	
	public function __construct($data = null)
	{
		$this->data = $data;
		$this->left = null;
		$this->right = null;
	}
}

### 树 ###
```
class Tree
{
	
    /**
     * 根节点
     */
	public $root = null;
	 
	public function __construct($headData = null)
	{
		if ($headData !=null) {
			$this->root = new TreeNode($headData);
		}
	}


    /**
     * 查找节点
     * @param int $data 数据
     * @return TreeNode|null 节点
     */
    public function find($data)
	{
		if($this->root == null) {
			return null;
		}
		$node = $this->root;
		while($node != null){
			if($node->data == $data){
				return $node;
			}elseif($data > $node->data){
				$node = $node->right;
			}else{
				$node = $node->left;
			}
		}
		return null;
	}

    /**
     * 查找最小节点
     * @param TreeNode|null $root 根节点
     * @return TreeNode $node 最小节点
     */
    public function findMin($root=null)
    {
	    $node = is_null($root) ? $this->root : $root;
	    while($node->left != null){
	        $node = $node->left;
        }
        return $node;
    }

    /**
     * 查找最大节点
     * @param TreeNode|null $root 根节点
     * @return TreeNode $node 最大节点
     */
    public function findMax($root=null)
    {
	    $node = is_null($root) ? $this->root : $root;
	    while($node->right != null){
	        $node = $node->right;
        }
        return $node;
    }


    /**
     * 插入数据
     * @param int $data 数据
     * @return bool
     */
    public function insert($data)
	{
		if($this->root == null){
			$this->root = new TreeNode($data);
			return true;
		}

		$node = $this->root;

		while($node != null){
			if($data > $node->data){
				if($node->right == null){
					$node->right = new TreeNode($data);
					return true;
				}
				$node = $node->right;
			}else{
				if($node->left == null){
					$node->left = new TreeNode($data);
					return true;
				}
				$node = $node->left;
			}
		}
	}

    /**
     * 删除节点
     * @param int $data 数据
     */
    public function delete($data)
    {
        //节点
        $node = $this->root;
        //父节点
        $pnode = null;
        //寻找要删除的节点
        while($node != null){
            if($node->data == $data){
                break;
            }elseif($data > $node->data){
                $pnode = $node;
                $node = $node->right;
            }else{
                $pnode = $node;
                $node = $node->left;
            }
        }
        //寻找后继节点
        if ($node->left == NULL || $node->right == NULL) {
            //如果待删除结点无子节点或只有一个子节点，则后继节点为待删除节点，$c = $node
            $cp = $pnode;
            $c = $node;
        } else {
            //如果待删除结点有两个子节点，$c置为$node的直接后继，以待最后将待删除结点的值换为其后继的值
            //寻找待删除节点右子树的最小值作为后继节点
            $cp = $node;
            $c = $node->right;
            //寻找最小节点
            while($c->left != null){
                $cp = $c;
                $c = $c->left;
            }
        }
        //$c子节点
        if ($c->left != NULL) {
            $s = $c->left;
        } else {
            $s = $c->right;
        }
        //将$c的子节点的父母结点置为$c的父母结点，此处$c只可能有1个子节点，因为如果$c有两个子节点，则c不可能是$node的直接后继
        if ($s != NULL) {
            $sp = $cp;
        }

        if ($cp == NULL) {
            //如果$c的父母为空，说明$c=$node是根节点，删除根节点后直接将根节点置为根节点的子节点，此处$node是根节点，且拥有两个子节点，则$c是$node的后继结点，$c的父母就不会为空，就不会进入这个if
            $this->root = $s;
        } else if ($c == $cp->left) {
            //如果$c是其父节点的左右子节点，则将$c父母的左右子节点置为$c的左右子节点
            $cp->left = $s;
        } else {
            $cp->right = $s;
        }

        //如果$c!=$node，说明$c是$node的后继结点，交换$c和$node的值
        if ($c != $node) {
            $node->data = $c->data;
        }
    }

    /**
     * 前序遍历
     * @param TreeNode $node
     */
    public function preOrder($node)
	{
		if($node == null){
			return ;
		}
		echo $node->data . '->';
        $this->preOrder($node->left);
        $this->preOrder($node->right);
	}

    /**
     * 中序遍历
     * @param TreeNode $node
     */
    public function inOrder($node)
    {
        if($node == null){
            return ;
        }
        $this->preOrder($node->left);
        echo $node->data . '->';
        $this->preOrder($node->right);
    }

    /**
     * 后序遍历
     * @param TreeNode $node
     */
    public function postOrder($node)
    {
        if($node == null){
            return ;
        }
        $this->preOrder($node->left);
        $this->preOrder($node->right);
        echo $node->data . '->';
    }

    /**
     * 层级遍历
     * @param TreeNode|null $root 根节点
     * @return array
     */
    public function levelOrder($root=null)
    {
        $root = is_null($root) ? $this->root : $root;
        $queue = [$root];
        for ($i = 0; $i < count($queue); $i++) {
            $node = $queue[$i];
            //入栈
            if ($node->left) {
                $queue[] = $node->left;
            }
            if ($node->right) {
                $queue[] = $node->right;
            }
        }
        return $queue;
    }
}
